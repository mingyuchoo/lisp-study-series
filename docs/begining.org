** LISP uses prefix notation
*** operators are written before their operands.
#+begin_src common-lisp
(+ (/ (* 60 9) 5) 32)
;; + is an operator
;; * is an operator
;; / is an operator
;; 9 is an operand
;; 5 is an operand
;; 60 is an operand
;; 32 is an operand
#+end_src
